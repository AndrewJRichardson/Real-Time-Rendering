306 void rtr::Device::DrawLineBresenham(const glm::vec3& start, const glm::vec3& end) {

308  int x = (int)start.x; //Store both values of both vectors
309  int y = (int)start.y; //so we don't have to cast them repeatedly in the following loop
310  int ex = (int)end.x;

314  int dy = std::abs((int)end.y - (int)start.y);
315

317

319  int iy; //increment and decrement is based on which direction the line is going
320 
321  if (x < end.x) {

323  }
324  else {
325  ix = -1;
326  }

329  iy = 1;
330  }
331  else {
332  iy = -1;
333  }
334 
335

337 
338  while (!complete) {
339  DrawPoint(glm::vec3(x, y, start.z), 0xff, 0xff, 0xff); //Draw the current point
340 
341  if (x == ex && y == ey) { //Exit if the vectors match
342  complete = true;
343  continue;
344  }
345 
346  //A difference of 0.5 is used in bresenham's algorithm to determine whether we should move the passive axis along
347  //since we are using integers we need to scale the error by 2 to account for the loss of precision
348  int tError = error * 2;
