261  const glm::vec3& end) {

263  // repeatedly in the loop

270  int distY = std::abs((int)end.y - (int)start.y);
271  int error = distX - distY;
272  int incX = -1;
273  int incY = -1;
274 
275  // Which way is the line going to be drawn
276  if (startX < end.x) {
277  incX = 1;
278  }
279  if (startY < end.y) {
280  incY = 1;
281  }
282 
283  bool complete = false;
284 
285  while (!complete) {
286  DrawPixel(glm::vec3(startX, startY, start.z), 0xff, 0xff, 0xff);
287 
288  if (startX == endX && startY == endY) {
289  complete = true;
290  continue;
291  }
292 
293  // Mult by 2 to account for precision loss of using int
294  // instead of floats
295  int tError = error * 2;
296  if (tError > -distY) {
